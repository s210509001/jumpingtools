Tampermonkey 脚本代码

// ==UserScript==
// @name         题号自动跳转工具
// @namespace    http://tampermonkey.net/
// @version      1.0
// @description  根据题号自动跳转到对应平台的原题页面
// @author       s210509001(wmz)
// @match        *://*/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    // 监听快捷键 Ctrl+Shift+P
    document.addEventListener('keydown', function(e) {
        if (e.ctrlKey && e.shiftKey && e.key === 'P') {
            e.preventDefault();
            showPrompt();
        }
    });

    // 显示输入提示框
    function showPrompt() {
        const problemId = prompt('请输入题号（如: CF123A, AT_abc140_a, P1001）:');
        if (problemId) {
            redirectToProblem(problemId);
        }
    }

    // 根据题号跳转
    function redirectToProblem(problemId) {
        const normalizedId = problemId.trim().toUpperCase();

        // 匹配Codeforces题号
        if (normalizedId.match(/^CF?[\s_]*\d+[A-Z]*$/i) || normalizedId.match(/^\d+[A-Z]*$/)) {
            let contestId = normalizedId.replace(/^CF?[\s_]*/i, '');
            window.open(`https://codeforces.com/problemset/problem/${contestId.slice(0, -1)}/${contestId.slice(-1)}`, '_blank');
        }
        // 匹配AtCoder题号
        else if (normalizedId.match(/^AT?[\s_]*[A-Z]+\d+_?[A-Z]/i) || normalizedId.match(/^[A-Z]+\d+_?[A-Z]/i)) {
            let contestId = normalizedId.replace(/^AT?[\s_]*/i, '');
            if (!contestId.includes('_')) {
                contestId = contestId.replace(/([A-Z]+)(\d+)([A-Z])/, '$1$2_$3');
            }
            window.open(`https://atcoder.jp/contests/${contestId.slice(0, -2)}/tasks/${contestId}`, '_blank');
        }
        // 匹配洛谷题号
        else if (normalizedId.match(/^(LUOGU|LG|P)[\s_]*\d+/i)) {
            let problemNum = normalizedId.replace(/^(LUOGU|LG|P)[\s_]*/i, '');
            window.open(`https://www.luogu.com.cn/problem/P${problemNum}`, '_blank');
        }
        else {
            alert('未识别题号格式，请参考支持的格式示例！');
        }
    }
})();

使用说明

1. 安装Tampermonkey浏览器扩展

2. 添加新脚本，复制上方代码到脚本编辑器中保存

3. 在任何页面使用快捷键 Ctrl+Shift+P 打开题号输入框

4. 输入题号后会自动跳转到对应题目页面
